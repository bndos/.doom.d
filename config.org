;; Terminal/container clipboard bridge.  This keeps `emacs -nw` inside Docker
;; connected to the host clipboard through Wayland first, then X11.
(defun bndos-terminal-clipboard--enabled-p ()
  (not (string= (or (getenv "BNDOS_TERMINAL_CLIPBOARD") "1") "0")))

(defun bndos-terminal-clipboard--wayland-p ()
  (and (getenv "WAYLAND_DISPLAY")
       (executable-find "wl-copy")
       (executable-find "wl-paste")))

(defun bndos-terminal-clipboard--x11-p ()
  (and (getenv "DISPLAY")
       (executable-find "xclip")))

(defun bndos-terminal-clipboard-set (text)
  (cond
   ((bndos-terminal-clipboard--wayland-p)
    (with-temp-buffer
      (insert text)
      (call-process-region (point-min) (point-max) "wl-copy" nil nil nil)))
   ((bndos-terminal-clipboard--x11-p)
    (with-temp-buffer
      (insert text)
      (call-process-region (point-min) (point-max) "xclip" nil nil nil "-selection" "clipboard"))))
  text)

(defun bndos-terminal-clipboard-get ()
  (cond
   ((bndos-terminal-clipboard--wayland-p)
    (with-temp-buffer
      (when (zerop (call-process "wl-paste" nil t nil "--no-newline"))
        (buffer-string))))
   ((bndos-terminal-clipboard--x11-p)
    (with-temp-buffer
      (when (zerop (call-process "xclip" nil t nil "-selection" "clipboard" "-o"))
        (buffer-string))))))

(defun bndos-terminal-clipboard-enable ()
  (when (and (bndos-terminal-clipboard--enabled-p)
             (not (display-graphic-p))
             (or (bndos-terminal-clipboard--wayland-p)
                 (bndos-terminal-clipboard--x11-p)))
    (setq select-enable-clipboard t
          interprogram-cut-function #'bndos-terminal-clipboard-set
          interprogram-paste-function #'bndos-terminal-clipboard-get)))

(bndos-terminal-clipboard-enable)
(add-hook 'after-init-hook #'bndos-terminal-clipboard-enable)

  (defun my-envrc->plist ()
    "Return envrc environment as a plist with keyword keys.
Falls back to `process-environment` if envrc export fails."
    (let* ((raw-env
            (cond
             ((fboundp 'envrc--export)
              (cdr (ignore-errors (envrc--export (projectile-project-root)))))
             (t nil)))
           (env-source (or raw-env
                           (mapcar (lambda (s)
                                     (let ((split (split-string s "=" t)))
                                       (cons (car split) (mapconcat #'identity (cdr split) "="))))
                                   process-environment))))
      (seq-mapcat

  "Open PATH with the OS default application."
  (interactive "fOpen externally: ")
  (pcase system-type

    ('windows-nt
     (if (fboundp 'w32-shell-execute)
         (w32-shell-execute "open" (convert-standard-filename path))

     (cl-destructuring-bind (prog . args) (my--linux-opener-and-args path)

  (defun my/markdown-latex-preview-svg (formula display)
    "Compile FORMULA to a cached SVG file and return its path."
    (unless (and (executable-find "latex") (executable-find "dvisvgm"))
      (user-error "latex and dvisvgm are required for formula previews"))
    (make-directory my/markdown-latex-preview-cache-dir t)
    (let* ((color (my/markdown-latex-preview-color))
           (hash (secure-hash 'sha256 (format "%s\n%s\n%s" color display formula)))
           (dir (expand-file-name hash my/markdown-latex-preview-cache-dir))
           (tex-file (expand-file-name "formula.tex" dir))
           (dvi-file (expand-file-name "formula.dvi" dir))
           (svg-file (expand-file-name "formula.svg" dir)))
      (unless (file-exists-p svg-file)
        (make-directory dir t)
        (with-temp-file tex-file
          (insert "\\documentclass{article}\n"
                  "\\usepackage[active,tightpage]{preview}\n"
                  "\\usepackage{amsmath,amssymb,xcolor}\n"
                  "\\pagestyle{empty}\n"
                  "\\begin{document}\n"
                  "\\begin{preview}\n"
                  (format "\\color[HTML]{%s}\n" color)
                  (if display
                      (format "\\[%s\\]\n" formula)
                    (format "$%s$\n" formula))
                  "\\end{preview}\n"
                  "\\end{document}\n"))
        (let ((default-directory dir))
          (unless (zerop (call-process "latex" nil nil nil
                                       "-interaction=nonstopmode"
                                       "-halt-on-error"
                                       tex-file))
            (user-error "latex failed for formula preview; see %s" dir))
          (unless (and (file-exists-p dvi-file)
                       (zerop (call-process "dvisvgm" nil nil nil
                                            "--no-fonts"
                                            "--exact-bbox"
                                            dvi-file
                                            "-o" svg-file)))
            (user-error "dvisvgm failed for formula preview; see %s" dir))))
      svg-file))

  (defun my/markdown-visual-table-align--common-indent (lines)
    "Return the indentation shared by every non-empty line in LINES."
    (let ((indents (delq nil
                         (mapcar (lambda (line)
                                   (unless (string-empty-p line)
                                     (if (string-match "\\`[ \t]*" line)
                                         (match-string 0 line)
                                       "")))
                                 lines))))
      (when indents
        (let ((prefix (car indents)))
          (dolist (indent (cdr indents))
            (while (not (string-prefix-p prefix indent))
              (setq prefix (substring prefix 0 (max 0 (1- (length prefix)))))))
          prefix))))

  (defun my/markdown-visual-table-align--aligned-lines (text)
    "Return markdown-mode aligned table lines for TEXT without editing this buffer."
    (let* ((lines (split-string (string-remove-suffix "\n" text) "\n"))
           (indent (or (my/markdown-visual-table-align--common-indent lines) ""))
           (stripped-lines
            (mapcar (lambda (line)
                      (if (and (not (string-empty-p indent))
                               (string-prefix-p indent line))
                          (substring line (length indent))
                        line))
                    lines)))
      (with-temp-buffer
        (dolist (line stripped-lines)
          (insert line "\n"))
        (goto-char (point-min))
        (let ((markdown-mode-hook nil))
          (markdown-mode))
        (when (or (markdown-table-at-point-p)
                  (looking-at-p "^[ \t]*|.*|[ \t]*$"))
          (let ((inhibit-message t))
            (condition-case nil
                (markdown-table-align)
              (error nil))))
        (mapcar (lambda (line) (concat indent line))
                (split-string
                 (string-remove-suffix
                  "\n" (buffer-substring-no-properties (point-min) (point-max)))
                 "\n")))))

  (defun my/markdown-visual-table-align--render-formulas-in-line (line)
    "Return LINE with inline math rendered as image display properties when possible."
    (if (and (fboundp 'my/markdown-latex-preview-svg)
             (display-images-p))
        (let ((rendered (copy-sequence line))
              (start 0))
          (while (string-match "\\(^\\|[^\\\\]\\)\\$\\([^$\n]+?\\)\\$" rendered start)
            (let* ((formula-start (+ (match-beginning 0)
                                     (length (match-string 1 rendered))))
                   (formula-end (match-end 0))
                   (formula (match-string 2 rendered)))
              (unless (string-prefix-p "$" formula)
                (condition-case nil
                    (let* ((svg (my/markdown-latex-preview-svg formula nil))
                           (image (create-image svg 'svg nil :ascent 'center :scale my/markdown-latex-preview-scale)))
                      (add-text-properties formula-start formula-end
                                           `(display ,image) rendered))
                  (error nil)))
              (setq start formula-end)))
          rendered)
      line))

  (defun my/markdown-visual-table-align--render-code-spans-in-line (line)
    "Return LINE with inline code spans styled for display-only table rows."
    (let ((rendered (copy-sequence line))
          (start 0)
          (face (if (facep 'markdown-code-face)
                    'markdown-code-face
                  'font-lock-constant-face)))
      (while (string-match "\\(^\\|[^\\\\]\\)`\\([^`\n]+\\)`" rendered start)
        (let* ((tick-start (+ (match-beginning 0)
                              (length (match-string 1 rendered))))
               (code-start (1+ tick-start))
               (tick-end (1- (match-end 0)))
               (code-end tick-end))
          (add-text-properties tick-start code-start '(display "") rendered)
          (add-text-properties code-start code-end `(face ,face) rendered)
          (add-text-properties tick-end (match-end 0) '(display "") rendered)
          (setq start (match-end 0))))
      rendered))

  (defun my/markdown-visual-table-align--pipe-targets (line)
    "Return display-column targets for pipe characters in LINE."
    (let ((pos 0)
          targets)
      (while (setq pos (string-match "|" line pos))
        (push (string-width (substring-no-properties line 0 pos)) targets)
        (setq pos (1+ pos)))
      (nreverse targets)))

  (defun my/markdown-visual-table-align--line-with-aligned-pipes (line)
    "Render formulas in LINE and force pipe separators to aligned columns."
    (let* ((targets (my/markdown-visual-table-align--pipe-targets line))
           (rendered (my/markdown-visual-table-align--render-code-spans-in-line
                      (my/markdown-visual-table-align--render-formulas-in-line line)))
           (pipe-positions nil)
           (pos 0))
      (while (setq pos (string-match "|" rendered pos))
        (push pos pipe-positions)
        (setq pos (1+ pos)))
      (setq pipe-positions (nreverse pipe-positions))
      (if (<= (length pipe-positions) 1)
          rendered
        (let ((out "")
              (prev (1+ (car pipe-positions)))
              (rest-pipes (cdr pipe-positions))
              (rest-targets (cdr targets)))
          (setq out (concat out (substring rendered 0 prev)))
          (while rest-pipes
            (let* ((pipe (car rest-pipes))
                   (target (car rest-targets))
                   (cell (string-trim-right (substring rendered prev pipe))))
              (setq out (concat out cell
                                (propertize " " 'display `(space :align-to ,target))
                                "|"))
              (setq prev (1+ pipe)
                    rest-pipes (cdr rest-pipes)
                    rest-targets (cdr rest-targets))))
          (concat out (substring rendered prev))))))

  (defun my/markdown-latex-preview-ranges (beg end)
    "Return TeX formula ranges between BEG and END, skipping table rows."
    (let (ranges)
      (save-excursion
        (goto-char beg)
        (while (search-forward "$$" end t)
          (let ((start (match-beginning 0)))
            (when (and (search-forward "$$" end t)
                       (not (my/markdown-visual-table-align--pipe-table-line-p)))
              (push (list start (match-end 0)
                          (string-trim (buffer-substring-no-properties
                                        (+ start 2) (- (match-end 0) 2)))
                          t)
                    ranges)))))
      (save-excursion
        (goto-char beg)
        (while (re-search-forward "\\\\\\[\\(?:.\\|\n\\)*?\\\\\\]" end t)
          (unless (or (my/markdown-range-overlaps-p (match-beginning 0) (match-end 0) ranges)
                      (save-excursion
                        (goto-char (match-beginning 0))
                        (my/markdown-visual-table-align--pipe-table-line-p)))
            (push (list (match-beginning 0) (match-end 0)
                        (string-trim (buffer-substring-no-properties
                                      (+ (match-beginning 0) 2)
                                      (- (match-end 0) 2)))
                        t)
                  ranges))))
      (save-excursion
        (goto-char beg)
        (while (re-search-forward "\\(^\\|[^\\\\]\\)\\$\\([^$\n]+?\\)\\$" end t)
          (let ((start (+ (match-beginning 0) (length (match-string 1))))
                (finish (match-end 0)))
            (unless (or (my/markdown-range-overlaps-p start finish ranges)
                        (string-prefix-p "$" (match-string 2))
                        (save-excursion
                          (goto-char start)
                          (my/markdown-visual-table-align--pipe-table-line-p)))
              (push (list start finish (match-string-no-properties 2) nil)
                    ranges)))))
      (sort ranges (lambda (a b) (< (car a) (car b))))))

(defun pr-review--collect-comment-positions ()
  "Return sorted positions of all comments: inline threads, pending, and review sections."
  (let (positions)
    (save-excursion
      ;; Inline diff thread links (pr-review-eldoc-content marks the summary line)
      (goto-char (point-min))
      (while-let ((m (text-property-search-forward
                      'pr-review-eldoc-content nil (lambda (_ v) v))))
        (push (prop-match-beginning m) positions))
      ;; Inline pending comment blocks
      (goto-char (point-min))
      (while-let ((m (text-property-search-forward
                      'pr-review-pending-review-thread nil (lambda (_ v) v))))
        (push (prop-match-beginning m) positions))
      ;; Bottom review thread sections
      (goto-char (point-min))
      (while-let ((m (text-property-search-forward
                      'magit-section nil
                      (lambda (_ v) (pr-review--review-thread-section-p v)))))
        (push (prop-match-beginning m) positions)
        ;; Skip to end of this section to avoid re-matching its interior
        (goto-char (prop-match-end m))))
    (sort (delete-dups positions) #'<)))

(defun pr-review--comment-skip-pos (pos direction)
  "Return a position past the comment at POS for DIRECTION (1 or -1)."
  (let ((section (get-text-property pos 'magit-section)))
    (cond
     ((pr-review--review-thread-section-p section)
      (if (> direction 0) (oref section end) (1- (oref section start))))
     ((get-text-property pos 'pr-review-pending-review-thread)
      (if (> direction 0)
          (or (next-single-property-change pos 'pr-review-pending-review-thread) pos)
        (1- (or (previous-single-property-change pos 'pr-review-pending-review-thread) (1+ pos)))))
     ((get-text-property pos 'pr-review-eldoc-content)
      (if (> direction 0)
          (or (next-single-property-change pos 'pr-review-eldoc-content) pos)
        (1- (or (previous-single-property-change pos 'pr-review-eldoc-content) (1+ pos)))))
     (t pos))))

(defun pr-review-next-comment ()
  "Move to the next inline or review-section comment."
  (interactive)
  (let* ((start (pr-review--comment-skip-pos (point) 1))
         (positions (pr-review--collect-comment-positions))
         (next (seq-find (lambda (p) (> p start)) positions)))
    (if next (goto-char next) (user-error "No next comment"))))

(defun pr-review-prev-comment ()
  "Move to the previous inline or review-section comment."
  (interactive)
  (let* ((start (pr-review--comment-skip-pos (point) -1))
         (positions (pr-review--collect-comment-positions))
         (prev (car (last (seq-filter (lambda (p) (< p start)) positions)))))
    (if prev (goto-char prev) (user-error "No previous comment"))))

(defun my/ein-stop-current-process ()
  "Stop EIN's current Jupyter process and clear stale process buffers."
  (when-let ((proc (ein:jupyter-server-process)))
    (let ((pid (process-id proc))
          (url (my/ein-current-url)))
      (if url
          (ignore-errors (ein:jupyter-server-stop nil url))
        (ignore-errors (delete-process proc)))
      (cl-loop repeat 20
               while (my/ein-jupyter-live-process-p)
               do (sleep-for 0 100))
      (when (my/ein-jupyter-live-process-p)
        (ignore-errors (delete-process proc))
        (when pid
          (ignore-errors (signal-process pid 9))))))
  (when-let ((buffer (get-buffer *ein:jupyter-server-buffer-name*)))
    (kill-buffer buffer)))

    (pcase-let ((`(,beg . ,end) (my/ein-markdown-edit-block-bounds)))
      (when (< beg end)
        (my/ein-clear-markdown-render-region beg end)
        (when (and my/ein-render-markdown-cell-align-source
                   (fboundp 'my/markdown-align-table-at-point-noerror))
          (save-excursion
            (goto-char beg)
            (when (re-search-forward "^[ \t]*|.*|[ \t]*$" end t)
              (my/markdown-align-table-at-point-noerror))))
        (when (fboundp 'my/markdown-render-style-overlays)
          (my/markdown-render-style-overlays beg end))
        (when (fboundp 'my/markdown-visual-table-align-refresh-region)
          (setq my/markdown-visual-table-align-editing nil)
          (my/markdown-visual-table-align-refresh-region beg end))
        (when (fboundp 'my/markdown-render-latex-previews-region)
          (my/markdown-render-latex-previews-region beg end)))))

  (defun my/ein-render-markdown-cell-region (cell)
    "Refresh visual markdown previews for CELL in the current notebook buffer."
    (when (fboundp 'my/markdown-render-style-overlays)
      (my/markdown-render-style-overlays
       (ein:cell-input-pos-min cell)
       (ein:cell-input-pos-max cell)))
    (when (and (fboundp 'my/markdown-visual-table-align-mode)
               (fboundp 'my/markdown-visual-table-align-refresh-region))
      ;; In the notebook buffer, never enable the table-align minor mode:
      ;; its after-change hook scans the whole notebook and can treat code
      ;; comments as markdown.  Render only this markdown cell's bounds.
      (when (bound-and-true-p my/markdown-visual-table-align-mode)
        (my/markdown-visual-table-align-mode -1))
      (when my/ein-render-markdown-cell-align-source
        (save-excursion
          (goto-char (ein:cell-input-pos-min cell))
          (when (re-search-forward "^[ \t]*|.*|[ \t]*$" (ein:cell-input-pos-max cell) t)
            (my/markdown-align-table-at-point-noerror))))
      (setq my/markdown-visual-table-align-editing nil)
      (my/markdown-visual-table-align-refresh-region
       (ein:cell-input-pos-min cell)
       (ein:cell-input-pos-max cell)))
    (let ((formula-count
           (my/markdown-render-latex-previews-region
            (ein:cell-input-pos-min cell)
            (ein:cell-input-pos-max cell)))
          (table-count (if (boundp 'my/markdown-visual-table-align-overlays)
                           (cl-count-if (lambda (overlay)
                                          (and (overlay-buffer overlay)
                                               (< (overlay-start overlay) (ein:cell-input-pos-max cell))
                                               (> (overlay-end overlay) (ein:cell-input-pos-min cell))))
                                        my/markdown-visual-table-align-overlays)
                         0)))
      (my/markdown-rendered-table-align-pipes-region
       (ein:cell-input-pos-min cell)
       (ein:cell-input-pos-max cell))
      (message "Refreshed %s table lines; rendered %s formula previews"
               table-count formula-count)))

  (defun my/ein-render-current-markdown-formulas ()
    "Refresh markdown table and formula previews in EIN markdown edit or notebook buffers."
    (interactive)
    (when (and (boundp 'my/markdown-html-render-overlays)
               my/markdown-html-render-overlays)
      (my/markdown-clear-html-render))
    (cond
     ((and (fboundp 'my/ein-python-cell-buffer-p)
           (my/ein-python-cell-buffer-p))
      (my/ein-clear-markdown-render-in-current-buffer)
      (user-error "Not in a markdown cell"))
     ((and (eq major-mode 'ein:markdown-mode)
           (buffer-base-buffer)
           (fboundp 'my/markdown-render-current-buffer)
           (fboundp 'my/markdown-render-latex-previews-region)
           (fboundp 'ein:worksheet-get-current-cell))
      (let ((cell (with-timeout (2 nil)
                    (ein:worksheet-get-current-cell :noerror t)))
            (notebook-buffer (buffer-base-buffer))
            (notebook-key (my/ein-buffer-notebook-key)))
        (unless (and cell (ein:markdowncell-p cell))
          (user-error "Not in a markdown cell"))
        (my/ein-clear-markdown-render-in-sibling-cell-buffers notebook-key)
        ;; Render only the current markdown block in the visible edit buffer.
        ;; Some EIN edit buffers contain the whole notebook text; whole-buffer
        ;; rendering makes Python comments look like markdown headings.
        (my/ein-render-markdown-edit-block)
        (with-current-buffer notebook-buffer
          (my/ein-render-markdown-cell-region cell))
        cell))
     ((and (derived-mode-p 'ein:markdown-mode 'markdown-mode)
           (fboundp 'my/markdown-render-current-buffer))
      (my/markdown-render-current-buffer))
     ((and (fboundp 'my/markdown-render-latex-previews-region)
           (fboundp 'ein:worksheet-get-current-cell))
      (let ((cell (with-timeout (2 nil)
                    (ein:worksheet-get-current-cell :noerror t))))
        (unless (and cell (ein:markdowncell-p cell))
          (user-error "Not in a markdown cell"))
        (my/ein-render-markdown-cell-region cell)
        cell))
     (t
      (user-error "Markdown rendering helpers are not available"))))

    "Return the owning EIN notebook buffer for the current edit/notebook buffer."
    (or (buffer-base-buffer)

               (trimmed (string-trim-left name))
               (cell-suffix (string-match "\[[^]]+\]\'" trimmed)))
          (when (and cell-suffix (string-prefix-p "*ein: " trimmed))
            (get-buffer (substring trimmed 0 cell-suffix))))
        (when (and (string-prefix-p "*ein: " (buffer-name))
                   (string-suffix-p "*" (buffer-name)))
          (current-buffer))))

  (defun my/ein-goto-next-input-noerror (&optional anchor-cell)
    "Move to the next EIN input block from notebook or cell edit buffers.
When ANCHOR-CELL is non-nil, move from that exact cell instead of the
notebook buffer's current ewoc node.  This keeps M-RET stable from indirect
markdown edit buffers whose point/current node can drift during rendering."
    (when (fboundp 'ein:worksheet-goto-next-input)
      (when-let* ((origin-buffer (current-buffer))
                  (notebook-buffer (my/ein-notebook-buffer-for-current-buffer)))
        (ignore-errors
          (with-timeout (1 nil)
            (unless (eq origin-buffer notebook-buffer)
              (set-window-buffer (selected-window) notebook-buffer)
              (my/ein--close-extra-edit-windows
               origin-buffer notebook-buffer (selected-window)))
            (when-let ((window (get-buffer-window notebook-buffer)))
              (select-window window))
            (with-current-buffer notebook-buffer
              (cond
               ((and anchor-cell (fboundp 'ein:cell-next) (fboundp 'ein:cell-goto))
                (if-let ((next-cell (ein:cell-next anchor-cell)))
                    (ein:cell-goto next-cell)
                  (message "No next input")))
               ((when-let ((node (ein:worksheet-get-current-ewoc-node)))
                  (ein:worksheet-goto-next-input node))))))))))

(defun my/ein-start-jupyter (&optional directory)
  "Start a managed Jupyter server on a free local port and open notebooks."
  (interactive
   (progn
     (my/ein-load)
     (list (read-directory-name "Notebook directory: "
                                ein:jupyter-default-notebook-directory
                                ein:jupyter-default-notebook-directory
                                t))))
  (my/ein-load)
  (unless (my/ein-managed-live-p)
    (my/ein-stop-current-process))
  (when-let ((buf (get-buffer *ein:jupyter-server-buffer-name*)))
    (let ((kill-buffer-query-functions nil))
      (kill-buffer buf)))
  (let* ((port (my/ein-free-local-port))
         (url (format "http://127.0.0.1:%s" port))
         (token (my/ein-random-token))
         (ein:jupyter-server-args
          (append ein:jupyter-server-args
                  (list (format "--IdentityProvider.token=%s" token)
                        (format "--NotebookApp.token=%s" token)))))
    (setq my/ein-managed-token token
          my/ein-managed-url url)
    (ein:jupyter-server-start
     (my/ein--jupyter-command)
     (file-name-as-directory (expand-file-name directory))
     t nil
     port)
    (cl-loop repeat 40
             until (my/ein-jupyter-live-process-p)
             do (sleep-for 0 100))
    (my/ein-login-managed-server)))

(defun my/display-current-buffer-in-side (side slot size)
  (interactive)
  (let ((buffer (current-buffer))
        (window (selected-window)))
    ;; `vterm' is usually already displayed in a dedicated side window via
    ;; `display-buffer-alist'.  Re-displaying that same live side-window buffer
    ;; tends to keep/reuse the existing window, so remove that window first.
    (when (window-live-p window)
      (set-window-dedicated-p window nil)
      (when (window-parameter window 'window-side)
        (delete-window window)))
    (select-window
     (display-buffer-in-side-window
      buffer
      `((side . ,side)
        (slot . ,slot)
        ,(if (memq side '(left right))
             `(window-width . ,size)
           `(window-height . ,size))
        (dedicated . side)
        (window-parameters . ((no-delete-other-windows . t))))))))

(defun my-make-room-for-new-compilation-buffer ()
